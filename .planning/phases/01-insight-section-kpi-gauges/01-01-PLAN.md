```javascript

  /* ── KPI 달성 게이지 ── */
  function drawGauge(canvasId, ratio, color){
    const cvs=document.getElementById(canvasId);
    if(!cvs)return;
    const ctx=cvs.getContext('2d');
    const dpr=window.devicePixelRatio||1;
    cvs.width=120*dpr;
    cvs.height=120*dpr;
    ctx.scale(dpr,dpr);
    const cx=60,cy=70,r=48,lw=10;
    const clampedRatio=Math.min(Math.max(ratio,0),1.5);
    const fillAngle=Math.PI+(clampedRatio*Math.PI);
    // track
    ctx.beginPath();
    ctx.arc(cx,cy,r,Math.PI,0,false);
    ctx.lineWidth=lw;
    ctx.lineCap='round';
    ctx.strokeStyle=getComputedStyle(document.documentElement).getPropertyValue('--border').trim()||'#e2e5ea';
    ctx.stroke();
    // filled arc
    if(ratio>0){
      ctx.beginPath();
      ctx.arc(cx,cy,r,Math.PI,Math.min(fillAngle,2*Math.PI),false);
      ctx.lineWidth=lw;
      ctx.lineCap='round';
      ctx.strokeStyle=color;
      ctx.stroke();
    }
    // center text
    const pct=(ratio*100).toFixed(0)+'%';
    ctx.font="bold 15px 'DM Mono',monospace";
    ctx.fillStyle=getComputedStyle(document.documentElement).getPropertyValue('--text').trim()||'#111827';
    ctx.textAlign='center';
    ctx.textBaseline='middle';
    ctx.fillText(pct,cx,cy-4);
  }

  // Render KPI gauges
  if(D.kpi&&D.kpi.items&&D.kpi.items.length>0){
    const accentColor=getComputedStyle(document.documentElement).getPropertyValue('--pink').trim()||'#2D2D6B';
    const gaugeIds=['gauge-rrms','gauge-ka','gauge-bld'];
    D.kpi.items.forEach((item,i)=>{
      const ratio=item.target>0?item.current/item.target:0;
      drawGauge(gaugeIds[i],ratio,accentColor);
    });
  }

```

**Key details:**
- Canvas uses 240x240 device pixels with CSS size 120x120 (2x DPI scaling per UI-SPEC)
- Semi-circle arc from PI to 0 (left to right, 180 degrees) per UI-SPEC
- Arc center at (60, 70) offset down to leave room for percentage text
- Arc radius 48px, line width 10px, lineCap round per UI-SPEC
- Track color from CSS var `--border`, filled arc from CSS var `--pink` (per UI-SPEC: Phase 1 uses accent only)
- Achievement percentage drawn at (cx, cy-4) in bold 15px DM Mono per UI-SPEC
- Ratio clamped to 0-1.5 for display (allows over-achievement to show visually beyond 100% but caps at 150%)
- Per requirement UI-02: section placed after Top10 grid (last section in buildOv)
- Per requirements GAUGE-01/02/03: each KPI item gets its own gauge with current/target display
- Empty state handled: shows "KPI 데이터가 없습니다" when no items
- Uses existing fmt() function for number formatting (e.g., 330 -> "330")
